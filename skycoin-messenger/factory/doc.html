▾ Example

This example shows how to send ping pong messages through two clients

Code:

package factory_test

import (
    "github.com/skycoin/net/skycoin-messenger/factory"
    "fmt"
    "github.com/skycoin/skycoin/src/cipher"
)

var client1pk cipher.PubKey

// This example shows how to send ping pong messages through two clients
func Example() {
    go startServer()
    go startClient1()
    startClient2()
}

func startServer() {
    f := factory.NewMessengerFactory()
    f.Listen(":8000")
}

func startClient1() {
    f := factory.NewMessengerFactory()

    err := f.Connect("localhost:8080")
    if err != nil {
        panic(err)
    }

    f.ForEachConn(processConn)
}

func processConn(conn *factory.Connection) {
    fmt.Println("connection")
    client1pk = conn.GetKey()
    for {
        select {
        case m, ok := <-conn.GetChanIn():
            if !ok {
                fmt.Println("got message: %s", string(m))
                panic("something is not ok...")
                return
            }
            fmt.Println("sending pong")
            conn.Write(factory.GenSendMsg(conn.GetKey(), conn.GetTargetKey(), []byte("pong")))
        }
    }
}

func startClient2() {
    f := factory.NewMessengerFactory()

    err := f.Connect("localhost:8080")
    if err != nil {
        panic(err)
    }

    f.ForEachConn(processConn2)
}

func processConn2(conn *factory.Connection) {
    conn.GetChanIn()
    factory.GenSendMsg(client1pk, conn.GetKey(), []byte("ping"))

}
